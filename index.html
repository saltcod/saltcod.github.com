Maintenance Mode

	function wpr_maintenance_mode() {
	    if ( !current_user_can( 'edit_themes' ) || !is_user_logged_in() ) {
	        wp_die('Maintenance, please come back soon.');
	    }
	}
	add_action('get_header', 'wpr_maintenance_mode');
